programa estadisticas;
vars
    n : entero;

entero suma_hasta(tope : entero) {
    vars
        i, acc : entero;
    {
        i = 1;
        acc = 0;
        mientras (i < tope + 1) haz
        {
            acc = acc + i;
            i = i + 1;
        };
        acc = acc;
    }
};

flotante promedio(total : entero, cant : entero) {
    vars
        r : flotante;
    {
        r = total / cant;
    }
};

nula reportar(tope : entero) {
    vars
        s : entero;
        p : flotante;
    {
        s = suma_hasta(tope);
        escribe("suma de 1 a", tope, "=", s);
        p = promedio(s, tope);
        escribe("promedio =", p);
    }
};

inicio
{
    n = 10;
    reportar(n);
    n = 5;
    reportar(n);
}
fin
